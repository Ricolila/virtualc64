54// Noise magic
55
56// The noise waveform of the SID is generated by a simple 23 bit LFSR.
57// On shifting, bit 0 is filled with bit 22 EXOR bit 17.
58#define NSHIFT(v, n) \
59(((v) << (n)) \
60| ((((v) >> (23 - (n))) ^ (v >> (18 - (n)))) & ((1 << (n)) - 1)))
61
62#define NVALUE(v) \
63(noiseLSB[v & 0xff] | noiseMID[(v >> 8) & 0xff] \
64| noiseMSB[(v >> 16) & 0xff])
65
66#define NSEED 0x7ffff8

289 // Returns the currently set oscillator frequency
290 u16 frequency() { return HI_LO(sidreg[0x01], sidreg[0x00]); }
291
292 /* Returns the pulse width of the pulse waveform
293 * The pulse width is a 12-bit number which linearly controls the pulse
294 * width (duty cycle) of the pulse waveform.
295 */
296 u16 pulseWidth() { return (u16)((sidreg[3] & 0x0F) << 8 | sidreg[0x02]); }
297
298 /* Returns the GATE bit for this voice
299 * The gate bit controls the Envelope Generator. When this bit is set to a
300 * one, the Envelope Generator is Gated (triggered) and the
301 * attack/decay/sustain cycle is initiated. When the bit is reset to a
302 * zero, the release cycle begins.
303 */
304 bool gateBit() { return sidreg[0x04] & 0x01; }
305
306 /* Returns the SYNC bit for this voice
307 * If this bit is set, hard sync effects are produced. Hard sync is where
308 * one waveform plays at its own frequency, but gets reset back to its
309 * start every time the second waveform loops. It is responsible for the
310 * rising modulating sound in Ben Daglish's Wilderness music from The Last
311 * Ninja and the ludicrous intro noise in Martin Galway's Roland's Rat Race
312 * music.
313 */
314 bool syncBit() { return (sidreg[0x04] & 0x02) != 0; }
315
316 /* Returns the RING MOD bit of the control register
317 * The RING MOD bit, when set to a one, replaces the Triangle waveform
318 * output of Oscillator 1 with a “Ring Modulated” combination of
319 * Oscillators 1 and 3.
320 */
321 bool ringModBit() { return (sidreg[0x04] & 0x04) != 0; }
322
323 /* Returns the TEST bit of the control register
324 * The TEST bit, when set to a one, resets and locks Oscillator 1 at zero
325 * until the TEST bit is cleared. The Noise waveform output of Oscillator 1
326 * is also reset and the Pulse waveform output is held at a DC level.
327 */
328 bool testBit() { return (sidreg[0x04] & 0x08) != 0; }
329
330 // Returns the waveform bits of the control register
331 u8 waveform() { return sidreg[0x04] & 0xF0; }
332
333 /* Returns the attack rate for the envelope generator
334 * The attack rate is a 4 bit value which determines how rapidly the output
335 * of the voice rises from zero to peak amplitude when the envelope
336 * generator is gated.
337 */
338 u8 attackRate() { return sidreg[0x05] >> 4; }
339
340 /* Returns the decay rate for the envelope generator
341 * The decay cycle follows the attack cycle and the decay rate determines
342 * how rapidly the output falls from the peak amplitude to the selected
343 * sustain level.
344 */
345 u8 decayRate() { return sidreg[0x05] & 0x0F; }
346
347 /* Returns the decay rate for the envelope generator
348 * The sustain cycle follows the decay cycle and the output of the voice
349 * will remain at the selected sustain amplitude as long as the gate bit
350 * remains set. The sustain levels range from zero to peak amplitude in 16
351 * linear steps, with a sustain value of 0 selecting zero amplitude and a
352 * sustain value of 15 selecting the peak amplitude. A sustain value of 8
353 * would cause the voice to sustain at an amplitude one-half the peak
354 * amplitude reached by the attack cycle.
355 */
356 u8 sustainRate() { return sidreg[0x06] >> 4; }
357
358 /* Returns the release rate for the envelope generator
359 * The release cycle follows the sustain cycle when the Gate bit is reset
360 * to zero. At this time, the output of Voice 1 will fall from the sustain
361 * amplitude to zero amplitude at the selected release rate. The 16 release
362 * rates are identical to the decay rates.
363 */
364 u8 releaseRate() { return sidreg[0x06] & 0x0F; }
365};
